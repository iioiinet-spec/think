💻 C 언어 구현
// ─── GCD & LCM 완전 구현 (C언어) ─── #include <stdio.h> /* 유클리드 호제법 (반복) */ int gcd(int a, int b) { while (b) { int t = b; b = a % b; a = t; } return a; } /* 재귀 버전 */ int gcd_rec(int a, int b) { return b == 0 ? a : gcd_rec(b, a % b); } /* LCM: a÷gcd 먼저 → 오버플로우 방지 */ long long lcm(int a, int b) { return (long long)a / gcd(a, b) * b; } int main() { int a, b; scanf("%d %d", &a, &b); int g = gcd(a, b); long long l = lcm(a, b); printf("GCD(%d, %d) = %d\n", a, b, g); printf("LCM(%d, %d) = %lld\n", a, b, l); /* 핵심 관계 검증 */ printf("\nA x B = %lld\n", (long long)a * b); printf("GCD x LCM = %lld\n", (long long)g * l); /* a', b' — gcd로 나눈 서로소 쌍 */ int ap = a / g; int bp = b / g; printf("\na' = A/GCD = %d, b' = B/GCD = %d\n", ap, bp); printf("LCM = GCD x a' x b' = %d x %d x %d = %lld\n", g, ap, bp, (long long)g * ap * bp); return 0; }
🐍 Python 구현
# ─── GCD & LCM 완전 구현 (Python) ─── import math # 방법 1: 직접 구현 (유클리드 호제법) def gcd(a, b): while b: a, b = b, a % b return a def lcm(a, b): return a // gcd(a, b) * b # a를 먼저 나눠 오버플로우 방지 # 방법 2: 내장 함수 (Python 3.5+ / 3.9+) # math.gcd(a, b) → Python 3.5+ # math.lcm(a, b) → Python 3.9+ def analyze(a, b): g = gcd(a, b) l = lcm(a, b) print(f"A = {a}, B = {b}") print(f"GCD({a}, {b}) = {g}") print(f"LCM({a}, {b}) = {l}") # ── 핵심 관계 ── print(f"\n[ 관계 검증 ]") print(f"A × B = {a * b}") print(f"GCD × LCM = {g * l}") print(f"같음? {a * b == g * l}") # ── 서로소 분해 ── ap, bp = a // g, b // g print(f"\n[ a', b' 분해 ]") print(f"a' = A÷GCD = {ap} (서로소)") print(f"b' = B÷GCD = {bp} (서로소)") print(f"LCM = GCD × a' × b' = {g} × {ap} × {bp} = {g*ap*bp}") print(f"GCD(a', b') = {gcd(ap, bp)} ← 반드시 1") a, b = map(int, input("두 수 입력 (공백 구분): ").split()) analyze(a, b)
