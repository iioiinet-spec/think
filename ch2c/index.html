🔢 소수(Prime Number)란?

1보다 큰 자연수 중 1과 자기 자신만을 약수로 갖는 수입니다.
예: 2, 3, 5, 7, 11, 13 …

소수 판별 핵심: 2 ≤ i ≤ √n 범위에서 나누어지면 소수가 아님
🖥 소수 판별 코드
// 소수 판별 (C언어) #include <stdio.h> #include <math.h> int isPrime(int n) { if (n < 2) return 0; for (int i = 2; i * i <= n; i++) { if (n % i == 0) return 0; } return 1; } int main() { int n; scanf("%d", &n); if (isPrime(n)) printf("%d은(는) 소수입니다\n", n); else printf("%d은(는) 소수가 아닙니다\n", n); return 0; }
# 소수 판별 (Python) import math def is_prime(n): if n < 2: return False for i in range(2, int(math.sqrt(n)) + 1): if n % i == 0: return False return True n = int(input()) if is_prime(n): print(f"{n}은(는) 소수입니다") else: print(f"{n}은(는) 소수가 아닙니다")

🧮 에라토스테네스의 체

1~N 범위의 소수를 한꺼번에 빠르게 찾는 알고리즘입니다.
소수의 배수를 차례로 지워나가면 남은 수들이 모두 소수!

시간복잡도: O(N log log N) — 단순 반복보다 훨씬 빠름
🖥 에라토스테네스의 체 코드
// 에라토스테네스의 체 (C언어) #include <stdio.h> #include <string.h> int main() { int N; scanf("%d", &N); int sieve[N+1]; memset(sieve, 1, sizeof(sieve)); sieve[0] = sieve[1] = 0; for (int i = 2; i * i <= N; i++) { if (sieve[i]) { for (int j = i*i; j <= N; j += i) sieve[j] = 0; } } printf("소수: "); for (int i = 2; i <= N; i++) if (sieve[i]) printf("%d ", i); return 0; }
# 에라토스테네스의 체 (Python) def sieve_of_eratosthenes(N): sieve = [True] * (N + 1) sieve[0] = sieve[1] = False i = 2 while i * i <= N: if sieve[i]: for j in range(i*i, N+1, i): sieve[j] = False i += 1 return [i for i in range(2, N+1) if sieve[i]] N = int(input()) print("소수:", sieve_of_eratosthenes(N))

➗ 약수(Divisors)

어떤 수 N을 나누어 떨어지게 하는 수의 집합입니다.
약수는 항상 쌍으로 존재합니다: d와 N/d

효율적 탐색: i = 1 → √N 까지만 확인 → O(√N)
🖥 약수 구하기 코드
// 약수 구하기 (C언어) #include <stdio.h> #include <stdlib.h> void getDivisors(int n) { int divs[1000], cnt = 0; for (int i = 1; i * i <= n; i++) { if (n % i == 0) { divs[cnt++] = i; if (i != n / i) divs[cnt++] = n / i; } } /* 정렬 후 출력 */ for (int i = 0; i < cnt-1; i++) for (int j = i+1; j < cnt; j++) if (divs[i] > divs[j]) { int t = divs[i]; divs[i] = divs[j]; divs[j] = t; } printf("%d의 약수(%d개): ", n, cnt); for (int i = 0; i < cnt; i++) printf("%d ", divs[i]); } int main() { int n; scanf("%d", &n); getDivisors(n); return 0; }
# 약수 구하기 (Python) import math def get_divisors(n): divs = [] for i in range(1, int(math.sqrt(n)) + 1): if n % i == 0: divs.append(i) if i != n // i: divs.append(n // i) return sorted(divs) n = int(input()) d = get_divisors(n) print(f"{n}의 약수({len(d)}개):", d)

🔗 최대공약수 GCD

두 수의 공통 약수 중 가장 큰 수입니다.
유클리드 호제법: GCD(a, b) = GCD(b, a mod b) — b = 0이면 a가 GCD

GCD(48, 18) → GCD(18, 12) → GCD(12, 6) → GCD(6, 0) = 6
🖥 GCD 코드 (유클리드 호제법)
// 최대공약수 GCD (C언어 - 유클리드 호제법) #include <stdio.h> int gcd(int a, int b) { while (b) { int t = b; b = a % b; a = t; } return a; } /* 재귀 버전 */ int gcd_rec(int a, int b) { return b == 0 ? a : gcd_rec(b, a % b); } int main() { int a, b; scanf("%d %d", &a, &b); printf("GCD(%d, %d) = %d\n", a, b, gcd(a, b)); return 0; }
# 최대공약수 GCD (Python - 유클리드 호제법) import math # 방법 1: 직접 구현 def gcd(a, b): while b: a, b = b, a % b return a # 방법 2: 재귀 def gcd_rec(a, b): return a if b == 0 else gcd_rec(b, a % b) # 방법 3: 내장 함수 # math.gcd(a, b) a, b = map(int, input().split()) print(f"GCD({a}, {b}) = {gcd(a, b)}")

📐 최소공배수 LCM

두 수의 공통 배수 중 가장 작은 수입니다.
GCD를 이용하면 간단하게 구할 수 있어요!

LCM(a, b) = a × b ÷ GCD(a, b)
🖥 LCM 코드
// 최소공배수 LCM (C언어) #include <stdio.h> int gcd(int a, int b) { while (b) { int t = b; b = a % b; a = t; } return a; } long long lcm(int a, int b) { return (long long)a / gcd(a, b) * b; /* a를 먼저 나눠 오버플로우 방지 */ } int main() { int a, b; scanf("%d %d", &a, &b); printf("GCD = %d\n", gcd(a, b)); printf("LCM = %lld\n", lcm(a, b)); return 0; }
# 최소공배수 LCM (Python) import math def gcd(a, b): while b: a, b = b, a % b return a def lcm(a, b): return a // gcd(a, b) * b a, b = map(int, input().split()) g = gcd(a, b) l = lcm(a, b) print(f"GCD({a},{b}) = {g}") print(f"LCM({a},{b}) = {l}") # Python 3.9+ 내장 # math.lcm(a, b)
결과가 여기에 표시됩니다
💡 왜 a를 먼저 나눌까? a × b를 먼저 계산하면 큰 수에서 오버플로우가 날 수 있어요. C에서는 (a / gcd) * b 순서가 안전해요!
