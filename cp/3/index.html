for 문 (횟수 반복)

정해진 횟수만큼 반복 · C의 3요소 vs 파이썬 range

1for 문의 구조한 줄에 3요소

C의 for는 한 줄에 3요소를 세미콜론(;)으로 나눠 써요. 파이썬은 range로 간단하게!

C 언어파이썬
for (int i = 1; i <= 5; i++) {
    printf("%d ", i);
}
// 시작; 조건; 변화		 for i in range(1, 6):
    print(i, end=" ")

# range(시작, 끝+1)
⚠️ 파이썬 range 끝값 주의! range(1, 6)은 1,2,3,4,5 — 6은 포함 안 됨!
1부터 N까지 원하면 range(1, N+1)로 써야 해요. 가장 흔한 실수랍니다!
2range 3가지 사용법파이썬 핵심
range 형태의미결과
range(5)0부터 4까지0 1 2 3 4
range(1, 6)1부터 5까지1 2 3 4 5
range(1, 10, 2)1부터 2씩 증가1 3 5 7 9
range(10, 0, -1)10부터 거꾸로10 9 8 ... 1
🖥 1부터 N까지 합 구하기
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int sum = 0;
    for (int i = 1; i <= n; i++) {
        sum += i;      // 1부터 n까지 누적
    }
    printf("1부터 %d까지 합: %d\n", n, sum);
    return 0;
}
# 1부터 N까지 합
n = int(input())
total = 0
for i in range(1, n + 1):
    total += i        # 누적
print(f"1부터 {n}까지 합: {total}")

# 참고: sum(range(1, n+1)) 로도 가능!
sum = 0누적할 변수를 0으로 시작 (중요!)
sum += i매 반복마다 i를 더해 쌓기
range(1, n+1)n을 포함하려면 n+1까지!
10을 넣으면 55 (1+2+...+10). 100도 넣어보세요!

while 문 (조건 반복)

조건이 참인 동안 계속 반복 · 횟수를 모를 때

1while 문의 구조조건이 참인 동안

while은 "~하는 동안"이라는 뜻이에요. 조건이 참인 동안 계속 반복해요.
C와 파이썬이 거의 똑같아요!

C 언어파이썬
int i = 1;
while (i <= 5) {
    printf("%d ", i);
    i++; // 꼭 변화!
}		 i = 1
while i <= 5:
    print(i, end=" ")
    i += 1 # 꼭 변화!
⚠️ while은 무한루프 위험이 커요! for는 변화(i++)가 구조에 들어있지만,
while은 직접 안에서 i를 바꿔줘야 해요. 안 그러면 영원히 멈추지 않아요!
🖥 N을 반으로 계속 나누기 (몇 번?)
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int count = 0;
    while (n > 1) {       // n이 1보다 큰 동안
        n /= 2;           // 반으로 나누기
        count++;          // 횟수 세기
    }
    printf("%d번 나눴습니다\n", count);
    return 0;
}
# N을 1이 될 때까지 반으로 나누기
n = int(input())
count = 0
while n > 1:            # n이 1보다 큰 동안
    n //= 2              # 정수 나눗셈 (몫)
    count += 1
print(f"{count}번 나눴습니다")
while (n > 1)몇 번 도는지 미리 모름 — while이 딱 맞음
n /= 2 (C) // n //= 2 (Py)파이썬은 정수 나눗셈에 // 사용!
16→4번, 100→6번. 2의 거듭제곱을 넣어보세요!

중첩 반복문 (이중 for)

반복문 안의 반복문 · 표·별찍기·구구단표

1반복문 속 반복문바깥 × 안쪽

중첩 반복문은 반복문 안에 또 반복문을 넣는 거예요.
바깥 반복 1번마다 안쪽 반복이 전부 실행돼요. 표(격자)를 만들 때 필수!

바깥 i=1→안쪽 j=1,2,3 전부 실행
바깥 i=2→안쪽 j=1,2,3 전부 실행 (다시!)
바깥 i=3→안쪽 j=1,2,3 전부 실행 (또!)

바깥이 3번, 안쪽이 3번이면 총 3×3 = 9번 실행돼요.

🖥 별(★) 삼각형 그리기
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {      // 바깥: 줄
        for (int j = 1; j <= i; j++) {  // 안쪽: 별 개수
            printf("*");
        }
        printf("\n");              // 한 줄 끝나면 줄바꿈
    }
    return 0;
}
# 별 삼각형
n = int(input())
for i in range(1, n + 1):      # 바깥: 줄
    for j in range(i):          # 안쪽: 별 개수
        print("*", end="")
    print()                    # 줄바꿈

# 참고: print("*" * i) 로 한 줄에!
바깥 for (i)줄 번호 — i번째 줄엔 별 i개
안쪽 for (j)i개의 별을 출력 (j를 i까지)
바깥 끝의 줄바꿈안쪽이 끝나면 다음 줄로!
5를 넣으면 별 5줄. 3, 7도 넣어보세요!

break / continue (반복 제어)

반복을 멈추거나 건너뛰기

1break와 continue의 차이멈춤 vs 건너뛰기
키워드동작비유
break반복문을 완전히 탈출"그만! 나가자" 🚪
continue이번 회차만 건너뛰고 다음 반복"이건 넘기고 다음!" ⏭️
💡 break는 반복문 밖으로 완전히 나가고, continue는 안쪽 남은 코드만 건너뛰고 반복은 계속해요.
C와 파이썬 모두 break, continue 단어가 똑같아요!
🖥 break — 특정 수를 찾으면 멈추기
#include <stdio.h>

int main() {
    int target;
    scanf("%d", &target);
    for (int i = 1; i <= 100; i++) {
        if (i == target) {
            printf("%d 찾음! (i=%d)\n", target, i);
            break;         // 찾았으니 반복 종료
        }
        printf("%d 확인...\n", i);
    }
    return 0;
}
# target을 찾으면 멈추기
target = int(input())
for i in range(1, 101):
    if i == target:
        print(f"{target} 찾음! (i={i})")
        break              # 반복 종료
    print(f"{i} 확인...")
5를 넣으면 1~4 확인 후 5에서 멈춰요

🖥 continue — 짝수만 건너뛰고 홀수만 출력
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        if (i % 2 == 0) {
            continue;      // 짝수면 건너뛰기
        }
        printf("%d ", i);   // 홀수만 출력됨
    }
    printf("\n");
    return 0;
}
# 홀수만 출력 (짝수는 건너뛰기)
n = int(input())

    if i % 2 == 0:
        continue          # 짝수면 건너뛰기
    print(i, end=" ")
