2arr[i][j] 읽는 법행 먼저, 열 나중

가장 헷갈리는 부분! arr[i][j]에서 i는 행(세로 위치), j는 열(가로 위치)이에요.

표현의미위 격자에서

⚠️ 행과 열 순서 주의! arr[i][j]는 arr[세로][가로] 순서예요.
좌표 (x, y)를 생각할 때 x가 가로라 헷갈리기 쉬워요. 배열에서는 보통 [행][열] = [y][x]로 씁니다!

🖥 격자를 입력받아 특정 칸 읽기
#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int arr[20][20];
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &arr[i][j]);
    int r, c;
    scanf("%d %d", &r, &c);
    printf("arr[%d][%d] = %d\n", r, c, arr[r][c]);
    return 0;
}
# 격자 입력받고 특정 칸 읽기
R, C = map(int, input().split())
arr = []
for _ in range(R):
    row = list(map(int, input().split()))
    arr.append(row)

r, c = map(int, input().split())
print(f"arr[{r}][{c}] = {arr[r][c]}")
이중 for로 입력행마다 열을 돌며 R×C개 값을 채움
Py: append(row)한 줄씩 리스트로 만들어 격자에 추가
arr[r][c]r행 c열의 한 칸을 정확히 읽음
1줄=행 열, 격자 값들, 마지막=읽을 위치

🔵 C
for (int i = 0; i < R; i++) for (int j = 0; j < C; j++) printf("%d ", arr[i][j]); // 한 행 끝나면 줄바꿈
🟢 Python
for i in range(R): for j in range(C): print(arr[i][j], end=" ") print() # 행 끝 줄바꿈 # 또는 for row in arr:
💡 파이썬은 for row in arr:로 행을 통째로 꺼낼 수도 있어요! 그 안에서 for x in row:로 각 칸을 봐요. 인덱스가 필요 없을 때 편해요.
2행별 합 vs 열별 합방향 주의

행별 합은 한 행의 가로줄을 더하고, 열별 합은 한 열의 세로줄을 더해요. 반복문 순서가 바뀌어요!

1행의 합 = 4+5+6 = 15 (가로)

💡 행별 합은 for i: for j: rowSum[i] += arr[i][j], 열별 합은 for j: for i: colSum[j] += arr[i][j]. 어느 인덱스를 고정하느냐가 핵심!
🖥 격자 전체 합과 각 행의 합
#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int arr[20][20], total = 0;
    for (int i = 0; i < R; i++) {
        int rowSum = 0;
        for (int j = 0; j < C; j++) {
            scanf("%d", &arr[i][j]);
            rowSum += arr[i][j];
            total += arr[i][j];
        }
        printf("%d행 합: %d\n", i, rowSum);
    }
    printf("전체 합: %d\n", total);
    return 0;
}
# 전체 합 + 각 행의 합
R, C = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(R)]

total = 0
for i, row in enumerate(arr):
    print(f"{i}행 합: {sum(row)}")  # sum!
    total += sum(row)
print(f"전체 합: {total}")
C: rowSum 초기화각 행 시작마다 0으로 리셋
Py: sum(row)한 행을 통째로 더함! 반복문 불필요
enumerate(arr)행 번호와 행을 짝으로 순회
3×3 격자. 각 행 합과 전체 합(=45)!

🔵 C
// 상, 하, 좌, 우 int dr[4] = {-1, 1, 0, 0}; int dc[4] = {0, 0, -1, 1}; for (int d = 0; d < 4; d++) { int nr = r + dr[d]; int nc = c + dc[d]; }
🟢 Python
# 상, 하, 좌, 우 dr = [-1, 1, 0, 0] dc = [0, 0, -1, 1] for d in range(4): nr = r + dr[d] nc = c + dc[d]
💡 dr[d], dc[d]를 현재 위치에 더하면 다음 칸(nr, nc)이 나와요! 이 4방향 탐색이 미로 찾기, BFS/DFS, 최단 경로의 핵심 뼈대예요.
2격자 밖으로 나가지 않기경계 검사

이동한 칸이 격자 안에 있는지 꼭 확인해야 해요! 안 그러면 배열 밖을 읽어 오류가 나요.

🔵 C — 범위 검사
if (nr >= 0 && nr < R && nc >= 0 && nc < C) { // 격자 안! 안전하게 사용 printf("%d ", arr[nr][nc]); }
🟢 Python — 범위 검사
if 0 <= nr < R and 0 <= nc < C: # 격자 안! (연쇄 비교 가능) print(arr[nr][nc])
⚠️ 경계 검사는 필수! nr >= 0(위쪽), nr < R(아래쪽), nc >= 0(왼쪽), nc < C(오른쪽) 네 조건을 모두 확인!
파이썬은 0 <= nr < R처럼 연쇄 비교로 짧게 쓸 수 있어요.
🖥 특정 칸의 상하좌우 이웃 값 구하기
#include <stdio.h>

int dr[4] = {-1,1,0,0};  // 상하좌우
int dc[4] = {0,0,-1,1};
char* name[4] = {"상","하","좌","우"};

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int arr[20][20];
    for (int i=0;i<R;i++)
      for (int j=0;j<C;j++) scanf("%d",&arr[i][j]);
    int r, c; scanf("%d %d", &r, &c);
    for (int d = 0; d < 4; d++) {
        int nr = r+dr[d], nc = c+dc[d];
        if (nr>=0 && nr<R && nc>=0 && nc<C)
            printf("%s: %d\n", name[d], arr[nr][nc]);
        else
            printf("%s: 벽\n", name[d]);
    }
    return 0;
}
# 상하좌우 이웃 구하기
dr = [-1,1,0,0]      # 상하좌우
dc = [0,0,-1,1]
name = ["상","하","좌","우"]

R, C = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(R)]
r, c = map(int, input().split())

for d in range(4):
    nr, nc = r+dr[d], c+dc[d]
    if 0 <= nr < R and 0 <= nc < C:
        print(f"{name[d]}: {arr[nr][nc]}")
    else:
        print(f"{name[d]}: 벽")
dr/dc 방향 배열상하좌우 4방향을 숫자로 표현
nr=r+dr[d]현재 위치에 방향을 더해 다음 칸
경계 검사격자 밖이면 "벽"으로 처리
가운데(1,1)=5의 이웃: 상2 하8 좌4 우6

1대각선 — i와 j의 관계i==j

정사각 격자의 주대각선은 행과 열이 같은 칸(i==j)이에요! 왼쪽 위 → 오른쪽 아래.

주대각선: arr[0][0], arr[1][1], arr[2][2]

주대각선 합 = 1+5+9 = 15. 반대쪽 부대각선은 i+j == N-1인 칸이에요 (3+5+7)!

🖥 정사각 격자의 두 대각선 합
#include <stdio.h>

int main() {
    int N; scanf("%d", &N);
    int arr[20][20];
    for (int i=0;i<N;i++)
      for (int j=0;j<N;j++) scanf("%d",&arr[i][j]);
    int main_d = 0, anti_d = 0;
    for (int i = 0; i < N; i++) {
        main_d += arr[i][i];        // i==j
        anti_d += arr[i][N-1-i];  // i+j==N-1
    }
    printf("주대각선: %d\n", main_d);
    printf("부대각선: %d\n", anti_d);
    return 0;
}
# 두 대각선 합
N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]

main_d = sum(arr[i][i] for i in range(N))
anti_d = sum(arr[i][N-1-i] for i in range(N))

print(f"주대각선: {main_d}")
print(f"부대각선: {anti_d}")
arr[i][i]주대각선 — 행과 열이 같음 (i==j)
arr[i][N-1-i]부대각선 — i+j가 N-1 (오른쪽 위→왼쪽 아래)
3×3. 주대각선 1+5+9=15, 부대각선 3+5+7=15
